Check for cached resources

Find potentially cached resources.

Code

    var resourceList = window.performance.getEntriesByType("resource");
    for (i= 0; i < resourceList.length; i++){
      var downloadTime = resourceList[i].responseEnd  - resourceList[i].responseStart;
      if (downloadTime <= 0.0){
        console.log('Resource ' + resourceList[i].name + ' was cached.');
      } else{
        console.log('Resource ' + resourceList[i].name + ' was NOT cached.');
      }
    }
