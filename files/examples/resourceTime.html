Measuring resource load time using Resource Timing

Code

    var resourceList = window.performance.getEntriesByType("resource");
    for (i = 0; i < resourceList.length; i++){
      if (resourceList[i].initiatorType == "img") {
        console.log('Load time of ' + resourceList[i].name + ' is ' + 
        (resourceList[i].responseEnd - resourceList[i].startTime + ' ms'));
      }
    }
